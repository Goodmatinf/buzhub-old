use std::io;

fn main() {

	let mut a = String::new();

	io::stdin().read_line(&mut a).expect("aac");

	let a: i32 = a.trim().parse().expect("aaa");

	let mut s = String::new();

	let mut d = 0;
	let mut e = a-1;
	let mut z = 1;

	for i in 0..a-1 {
		for w in 0..e {
			s = format!("{}{}", s, ".");
		}
		for j in 0..z {
			s = format!("{}{}", s, "x");
		}
		for g in 0..e {
			s = format!("{}{}", s, ".");
		}
		println!("{}", s);
		s = String::new();
		z += 2;
		e -= 1;
	}

	for i in 0..a {
		for w in 0..d {
			s = format!("{}{}", s, ".");
		}
		for j in 0..z {
			s = format!("{}{}", s, "x");
		}
		for g in 0..d {
			s = format!("{}{}", s, ".");
		}
		println!("{}", s);
		s = String::new();
		z -= 2;
		d += 1;
	}
}
