Rozwiązanie w języku Rust: 
use std::io;

fn main() {

let mut a = String::new();

io::stdin().read_line(&mut a);

//let mut a = a.split(":");
let v: Vec<&str> = a.split(":").collect();
let mut m: i32 = v[0].trim().parse().expect("mmm");
let mut z: i32 = v[1].trim().parse().expect("zzz");
let mut s: i32 = v[2].trim().parse().expect("sss");
let mut d: i32 = v[3].trim().parse().expect("ddd");

if d > 11 {
let mut y = d%12;
let mut c = (d-y) / 12;
s += c;
d -= 12*c;
}

if s > 19 {
let mut y = s%20;
let mut c = (s-y) / 20;
z += c;
s -= 20*c;
}

if z > 99 {
let mut y = z%100;
let mut c = (z-y) / 100;
m += c;
z -= 100*c;
}

println!("{}:{}:{}:{}", m, z, s, d);
}
